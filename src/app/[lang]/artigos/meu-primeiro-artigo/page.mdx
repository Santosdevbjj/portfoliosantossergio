import MdxLayout from '@/components/mdx-layout'
import { getServerDictionary } from "@/lib/getServerDictionary"
import { normalizeLocale } from "@/dictionaries/locales"

export default async function Page(props: { params: Promise<{ lang: string }> }) {
  const { lang } = await props.params;
  const dict = await getServerDictionary(lang);
  const safeLang = normalizeLocale(lang);

  return (
    <MdxLayout lang={safeLang} dict={dict}>
      {/* O conteúdo abaixo é processado pelo Typography Engine do Tailwind 4.2 
          configurado no seu MdxLayout.tsx
      */}
      
      # {dict.articles.items[0]?.title || "Meu Primeiro Artigo Técnico"}

      > {dict.articles.items[0]?.description || "Este é um exemplo de artigo escrito manualmente."}

      ## Por que usar MDX manual?

      - **Controle Total**: Integre componentes React 19 diretamente no texto.
      - **Performance Máxima**: O Next.js 16 pré-renderiza este conteúdo no servidor.
      - **SEO Nativo**: Totalmente compatível com os metadados do seu dicionário.

      ---

      ## Exemplo de Código

      ```javascript

      console.log("Sistema operando em Node 24");
      ```

      ### Conclusão

      Este artigo está integrado ao seu sistema de **i18n** e utiliza o `TableOfContents` para gerar o sumário automaticamente a partir destes títulos (H2 e H3).

    </MdxLayout>
  );
}
