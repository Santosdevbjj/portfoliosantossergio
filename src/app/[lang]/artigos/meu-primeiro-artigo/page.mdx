import MdxLayout from '@/components/mdx-layout'
import { getServerDictionary } from "@/lib/getServerDictionary"
import { normalizeLocale } from "@/dictionaries/locales"

export const dynamic = 'force-static';

export default async function Page(props) {
  const params = await props.params;
  const lang = params.lang;
  const dict = await getServerDictionary(lang);
  const safeLang = normalizeLocale(lang);

  // Extração segura para evitar erros de renderização no servidor
  const article = dict.articles.items[0];
  const title = article?.title || "Artigo";
  const description = article?.description || "";

  return (
    <MdxLayout lang={safeLang} dict={dict}>
      <header className="mb-10">
        <h1 id="titulo-principal">{title}</h1>
        <blockquote className="border-l-4 border-blue-600 pl-4 italic text-slate-400">
          {description}
        </blockquote>
      </header>

      ## Por que usar MDX manual?

      O uso de MDX permite unir a simplicidade do Markdown com o poder do **React 19**. No contexto do seu portfólio, isso garante:

      - **Controle Total**: Integre componentes diretamente no texto.
      - **Performance Máxima**: O Next.js 16 pré-renderiza este conteúdo no servidor usando o motor Turbopack.
      - **SEO Nativo**: Totalmente compatível com os metadados do seu dicionário.

    </MdxLayout>
  );
}
